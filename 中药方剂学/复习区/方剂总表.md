# 方剂汇总
```dataviewjs
// 1. 获取全量方剂数据
let pages = dv.pages('"中药方剂学/方剂库"').where(p => p.type === "方剂");

// 2. 按“大类”进行第一级分组
let groupByClass = pages.groupBy(p => p.大类);

for (let bigClass of groupByClass) {
    dv.header(2, bigClass.key || "未分类大类");
    
    // 在大类下，按“子类”进行第二级分组
    let groupBySubClass = bigClass.rows.groupBy(p => p.子类);
    
    for (let subClass of groupBySubClass) {
        dv.header(3, subClass.key || "未分类子类");
        
        // 渲染该子类下的方剂表格
        dv.table(
            ["方剂名", "方剂组成", "配伍结构"],
            subClass.rows.map(p => {
                let name = p.file.link;

                let allIngredients = [];
                if (p.组成) {
                    allIngredients = dv.array(p.组成);
                } else {
                    if (p.君药) allIngredients.push(...dv.array(p.君药));
                    if (p.臣药) allIngredients.push(...dv.array(p.臣药));
                    if (p.佐药) allIngredients.push(...dv.array(p.佐药));
                    if (p.使药) allIngredients.push(...dv.array(p.使药));
                }
                let ingredientsStr = allIngredients.filter(x => x).join("、");

                let roles = [];
                const isValid = (arr) => arr && dv.array(arr).filter(x => x).length > 0;
                
                if (isValid(p.君药)) roles.push(`君:${dv.array(p.君药).filter(x=>x).join("、")}`);
                if (isValid(p.臣药)) roles.push(`臣:${dv.array(p.臣药).filter(x=>x).join("、")}`);
                if (isValid(p.佐药)) roles.push(`佐:${dv.array(p.佐药).filter(x=>x).join("、")}`);
                if (isValid(p.使药)) roles.push(`使:${dv.array(p.使药).filter(x=>x).join("、")}`);
                
                let rolesStr = roles.length > 0 ? "<ul style='margin: 0; padding-left: 20px;'>" + roles.map(r => "<li>" + r + "</li>").join("") + "</ul>" : "不分特定配伍";
                
                return [name, ingredientsStr || "未填写", rolesStr];
            })
        );
    }
}
```
